// Карусель команды
const teamCarousels = {};
let teamPerView = 3; // По умолчанию 3 карточки

function updateTeamPerView() {
  const width = window.innerWidth;
  if (width <= 768) {
    teamPerView = 1;
  } else if (width <= 1024) {
    teamPerView = 2;
  } else {
    teamPerView = 3;
  }
}

function initTeamCarousel(restaurantId, count) {
  teamCarousels[restaurantId] = { current: 0, total: count };
  const maxSlides = Math.ceil(count / teamPerView);
  const dotsContainer = document.getElementById(`team-dots-${restaurantId}`);
  if (!dotsContainer) return;
  dotsContainer.innerHTML = "";
  for (let i = 0; i < maxSlides; i++) {
    const dot = document.createElement("span");
    dot.className = `team-carousel-dot ${i === 0 ? "active" : ""}`;
    dot.onclick = () => teamGoTo(restaurantId, i);
    dotsContainer.appendChild(dot);
  }
  updateTeamCarousel(restaurantId);
}

function updateTeamCarousel(restaurantId) {
  const carousel = teamCarousels[restaurantId];
  if (!carousel) return;
  const track = document.getElementById(`team-track-${restaurantId}`);
  const cardWidth = 100 / teamPerView;
  const offset = -(carousel.current * cardWidth);
  if (track) track.style.transform = `translateX(${offset}%)`;
  const dots = document.querySelectorAll(`#team-dots-${restaurantId} .team-carousel-dot`);
  dots.forEach((dot, i) => dot.classList.toggle("active", i === carousel.current));
}

function teamNext(restaurantId) {
  const carousel = teamCarousels[restaurantId];
  if (!carousel) return;
  const maxSlide = Math.ceil(carousel.total / teamPerView) - 1;
  if (carousel.current < maxSlide) {
    carousel.current++;
    updateTeamCarousel(restaurantId);
  }
}

function teamPrev(restaurantId) {
  const carousel = teamCarousels[restaurantId];
  if (!carousel) return;
  if (carousel.current > 0) {
    carousel.current--;
    updateTeamCarousel(restaurantId);
  }
}

function teamGoTo(restaurantId, index) {
  const carousel = teamCarousels[restaurantId];
  if (!carousel) return;
  carousel.current = index;
  updateTeamCarousel(restaurantId);
}

window.addEventListener("resize", () => {
  updateTeamPerView();
  initTeamCarousel(1, 2);
  initTeamCarousel(2, 4);
  initTeamCarousel(3, 8);
  initTeamCarousel(4, 4);
  initTeamCarousel(5, 3);
  initTeamCarousel(6, 1);
});

if (typeof window !== "undefined") {
  window.addEventListener("DOMContentLoaded", () => {
    updateTeamPerView();
    initTeamCarousel(1, 2);
    initTeamCarousel(2, 4);
    initTeamCarousel(3, 8);
    initTeamCarousel(4, 4);
    initTeamCarousel(5, 3);
    initTeamCarousel(6, 1);
  });
}
