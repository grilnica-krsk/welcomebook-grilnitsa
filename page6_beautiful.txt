  <script>
// Карусель бонусов
let benefitsCurrentSlide = 0;
const benefitsCards = document.querySelectorAll('.benefit-card');
const benefitsTotal = benefitsCards.length;
let benefitsPerView = 3; // По умолчанию 3 карточки

// Определяем количество видимых карточек в зависимости от ширины экрана
function updateBenefitsPerView() {
  const width = window.innerWidth;
  if (width <= 768) {
    benefitsPerView = 1;
  } else if (width <= 1024) {
    benefitsPerView = 2;
  } else {
    benefitsPerView = 3;
  }
}

function updateBenefitsCarousel() {
  const track = document.getElementById('benefits-track');
  const cardWidth = 100 / benefitsPerView;
  const offset = -(benefitsCurrentSlide * cardWidth);
  track.style.transform = `translateX(${offset}%)`;

  // Обновляем индикаторы
  const maxSlide = Math.ceil(benefitsTotal / benefitsPerView) - 1;
  const indicators = document.querySelectorAll('.carousel-indicator');
  indicators.forEach((ind, i) => {
    ind.classList.toggle('active', i === benefitsCurrentSlide);
  });
}

function benefitsCarouselNext() {
  const maxSlide = Math.ceil(benefitsTotal / benefitsPerView) - 1;
  benefitsCurrentSlide = (benefitsCurrentSlide + 1) % (maxSlide + 1);
  updateBenefitsCarousel();
}

function benefitsCarouselPrev() {
  const maxSlide = Math.ceil(benefitsTotal / benefitsPerView) - 1;
  benefitsCurrentSlide = (benefitsCurrentSlide - 1 + maxSlide + 1) % (maxSlide + 1);
  updateBenefitsCarousel();
}

// Инициализация индикаторов
function initBenefitsCarousel() {
  updateBenefitsPerView();
  const maxSlide = Math.ceil(benefitsTotal / benefitsPerView);
  const indicatorsContainer = document.getElementById('benefits-indicators');
  indicatorsContainer.innerHTML = '';

  for (let i = 0; i < maxSlide; i++) {
    const indicator = document.createElement('span');
    indicator.className = `carousel-indicator ${i === 0 ? 'active' : ''}`;
    indicator.onclick = () => {
      benefitsCurrentSlide = i;
      updateBenefitsCarousel();
    };
    indicatorsContainer.appendChild(indicator);
  }

  updateBenefitsCarousel();
}

// Пересчитываем при изменении размера окна
window.addEventListener('resize', () => {
  updateBenefitsPerView();
  benefitsCurrentSlide = 0; // Сбрасываем позицию
  initBenefitsCarousel();
});

// Инициализация при загрузке
if (typeof window !== 'undefined') {
  window.addEventListener('DOMContentLoaded', initBenefitsCarousel);
}
  </script>
