const carousels = {};

function initCarousel(restaurantId, count) {
  carousels[restaurantId] = { current: 0, total: count };
  const dotsContainer = document.getElementById(`dots-${restaurantId}`);
  if (!dotsContainer) return;
  dotsContainer.innerHTML = "";
  for (let i = 0; i < count; i++) {
    const dot = document.createElement("span");
    dot.className = `carousel-dot ${i === 0 ? "active" : ""}`;
    dot.onclick = () => goToSlide(restaurantId, i);
    dotsContainer.appendChild(dot);
  }
  updateCarousel(restaurantId);
}

function updateCarousel(restaurantId) {
  const carousel = carousels[restaurantId];
  if (!carousel) return;
  const { current } = carousel;
  const track = document.getElementById(`track-${restaurantId}`);
  const dots = document.querySelectorAll(`#dots-${restaurantId} .carousel-dot`);
  if (track) track.style.transform = `translateX(-${current * 100}%)`;
  dots.forEach((dot, i) => dot.classList.toggle("active", i === current));
}

function nextSlide(restaurantId) {
  const carousel = carousels[restaurantId];
  if (!carousel) return;
  carousel.current = (carousel.current + 1) % carousel.total;
  updateCarousel(restaurantId);
}

function prevSlide(restaurantId) {
  const carousel = carousels[restaurantId];
  if (!carousel) return;
  carousel.current = (carousel.current - 1 + carousel.total) % carousel.total;
  updateCarousel(restaurantId);
}

function goToSlide(restaurantId, index) {
  const carousel = carousels[restaurantId];
  if (!carousel) return;
  carousel.current = index;
  updateCarousel(restaurantId);
}

if (typeof window !== "undefined") {
  window.addEventListener("DOMContentLoaded", () => {
    initCarousel(1, 2);
    initCarousel(2, 4);
    initCarousel(3, 8);
    initCarousel(4, 4);
    initCarousel(5, 3);
    initCarousel(6, 1);
  });
}
  </script>
